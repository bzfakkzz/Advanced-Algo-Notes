[P1518 - [蓝桥杯2020初赛] 回文日期 - New Online Judge (ecustacm.cn)](http://oj.ecustacm.cn/problem.php?id=1518)



分析：



枚举年份，制造合法回文串，

check回文串是否满足要求，如果满足要求就作为答案



`stoi(s)`能返回当前字符串代表的数值

如：`stoi("1234")==1234`

#include <bits/stdc++.h>

#define alls(x) x.begin(),x.end()

using namespace std;

int day[] = {
	31,28,31,30,31,30,31,31,30,31,30,31
};

string s, ans1, ans2;
string ss, tt;
int y, m, d;

bool check(int x)  // 检查是否为闰年
{
	return x % 4 == 0 && x % 100 || x % 400 == 0;
}

bool checkk(string s)  // 检查是否ABABBABA
{
	if (s[0] == s[2] && s[0] == s[5] && s[0] == s[7])
	{
		if (s[1] == s[3] && s[1] == s[4] && s[1] == s[6])return true;
	}

	return false;
}

void solve()
{
	cin >> s;

	ans1 = "", ans2 = "";

	for (int i = stoi(s.substr(0, 4)); ans1 == "" || ans2 == ""; i++)  // 枚举年份制造回文字符串
	{
		ss = to_string(i), tt = to_string(i);

		reverse(alls(tt));

		ss += tt;  // 制造回文字符串

		if (ss <= s)continue;  // 在当前日期之后

		y = stoi(ss.substr(0, 4)), m = stoi(ss.substr(4, 2)), d = stoi(ss.substr(6, 2));

		if (check(y))day[1] = 29;
		else day[1] = 28;

		if (m < 1 || m>12)continue;  // 不合法

		if (d<1 || d>day[m - 1])continue;

		if (ans1 == "")ans1 = ss;

		if (checkk(ss) && ans2 == "")ans2 = ss;
	}

	cout << ans1 << endl << ans2 << endl;
}

int main()
{
	int t;

	cin >> t;

	while (t--)solve();

	return 0;
}

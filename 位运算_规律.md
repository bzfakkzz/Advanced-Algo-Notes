[0异或变换 - 蓝桥云课 (lanqiao.cn)](https://www.lanqiao.cn/problems/1592/learning/)



问经过t次变换后得到的01串



分析：



$2^x$是恰好$\ge n$的数，以$2^x$次为一个循环节







```cpp
void solve()
{
    int cur=1;

    cin>>n>>t>>s;

    while(cur<n)
    {
        cur<<=1;
    }

    t%=cur;

    while(t--)
    {
        ss=s;

        for(int i=1;i<s.size();i++)
        {
            s=(ss[i-1]-'0')^(ss[i]-'0')+'0';
        }
    }

    cout<<s<<endl;
}
```



```cpp
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int n;
ll t, cur;
string s, ss;

bool check(string s)
{
    int cnt = 0;

    for (auto u : s)if (u == '0')cnt++;

    return cnt == n;
}

int main()
{
    cin >> n >> t >> s;

    if (check(s))
    {
        cout << s << endl;

        return 0;
    }

    cur = 1;

    while (cur < n)cur <<= 1;  // 循环节

    t %= cur;  // 第几个

    while (t--)
    {
        ss = s;

        for (int i = 1; i < n; i++)
        {
            s[i] = (ss[i - 1] - '0') ^ (ss[i] - '0') + '0';
        }
    }

    cout << s << endl;

    return 0;
}
```

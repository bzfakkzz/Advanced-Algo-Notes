[Problem - B - Codeforces](https://codeforces.com/contest/2021/problem/B)



分析：



将数组a从小到大排序，发现不能更新mex的就`mp[a[i]%x].insert(a[i])`，

`while(get(mex))mex++;`

每当`cnt[u]==0`就`mp[a[i]%x].erase(u)`



成功每次都删掉一个数，$O(10^{5})$



```cpp
bool find(int c)
{
    int cur=0;
    bool f=false;

    for(auto u:mp[c%x])
    {
        if(u>c)break;

        f=true;

        cur=u;
    }

    if(f)
    {
        cnt[cur]--;

        if(cnt[cur]==0)mp[c%x].erase(cur);
    }

    return f;
}

void solve()
{
    cin>>n>>x;

    mex=0;

    mp.clear();
    cnt.clear();

    for(int i=1;i<=n;i++)cin>>a[i];

    sort(a+1,a+n+1);

    for(int i=1;i<=n;i++)
    {
        if(a[i]==mex)mex++;
        else
        {
            mp[a[i]%x].insert(a[i]);
            cnt[a[i]]++;
        }
    }

    if(mex==n)cout<<mex<<endl;
    else
    {
        while(find(mex))mex++;

        cout<<mex<<endl;
    }
}
```
